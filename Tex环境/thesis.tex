%==================== 文档类 ====================
% ctexart：中文文章文档类，自动处理中文版式
% zihao=-4：正文使用 -4 号字（约 12pt）
% a4paper：A4 纸张
\documentclass[zihao=-4, a4paper]{ctexart}

%==================== 数学符号与公式 ====================
\usepackage{amsmath}      % 提供数学环境（如 equation、align）、数学符号增强
\usepackage{amsthm}       % 提供证明环境（proof）和定理样式（如不需要可注释）
\usepackage{amsfonts}     % 提供数学字体（如 \mathbb、\mathfrak）
\usepackage{bm}           % 提供数学粗体命令 \bm，用于加粗向量/矩阵

%==================== 页面边距 ====================
\usepackage{geometry}
\geometry{top=3.0cm, bottom=2.7cm, left=2.5cm, right=2.5cm}

%==================== 页眉页脚 ====================
\usepackage{fancyhdr}
\pagestyle{fancy}         % 启用 fancy 样式
\fancyhf{}                % 清空所有页眉页脚
\fancyhead[C]{\zihao{5}\kaishu 学号~姓名~BSMA Homework}  % 页眉居中
\fancyfoot[C]{~\zihao{5}\thepage~}                           % 页脚居中（页码）
\renewcommand{\headrulewidth}{0.65pt}  % 页眉横线粗细
\setlength{\headheight}{9pt}	%避免页眉与正文重叠。
%==================== 章节标题格式（ctexart 内置，可选调整） ====================
\ctexset{
  section = {
    format = {\centering\bfseries\zihao{-2}},  % 一级标题：居中、黑体、三号
    name = {第, 章}                            % 标题前缀：第 X 章
  },
  subsection = {
    nameformat = {\bfseries\zihao{3}}          % 二级标题：黑体、四号
  },
  subsubsection = {
    nameformat = {\bfseries\zihao{4}}          % 三级标题：黑体、小四
  }
}

%==================== 图表编号方式 ====================
% 以下命令让公式、图、表的编号包含章节号，如 (1.1)、图 2.3
\numberwithin{equation}{section}
\numberwithin{figure}{section}
\numberwithin{table}{section}

%==================== 图形与颜色 ====================
\usepackage{graphicx}     % 插入图片，提供 \includegraphics
\usepackage{xcolor}       % 颜色支持，已包含 color 宏包，可自定义颜色
\usepackage{caption}      % 自定义图表标题格式
\captionsetup{figurewithin=section}  % 让 figure 编号包含章节（与 numberwithin 作用相同，可省略其一）
\usepackage{float}		% allow [H].etc
%==================== 超链接与交叉引用 ====================
\usepackage{hyperref}
\hypersetup{hidelinks}    % 取消链接边框（默认红色框）

%==================== 代码高亮（listings） ====================
\usepackage{listings}
% 颜色定义（用于代码高亮）
\definecolor{dkgreen}{rgb}{0,0.6,0}
\definecolor{gray}{rgb}{0.5,0.5,0.5}
\definecolor{mauve}{rgb}{0.58,0,0.82}
% listings 全局设置
\lstset{
  numbers=left,                     % 行号左侧显示
  numberstyle=\color[RGB]{0,192,192}, % 行号颜色
  backgroundcolor=\color[RGB]{245,245,244}, % 代码背景色
  keywordstyle=\bf\color{blue},     % 关键字样式
  commentstyle=\it\color[RGB]{0,96,96}, % 注释样式
  stringstyle=\rmfamily\slshape\color[RGB]{128,0,0}, % 字符串样式
  showstringspaces=false,           % 不显示字符串中的空格标记
  framexleftmargin=8mm,             % 左边距
  frame=none                        % 无边框
}

%==================== MATLAB / Python 代码高亮风格 ====================
% 这两个包提供专用的代码环境（如 \begin{MATLAB}、\begin{python}）
\usepackage{MATLAB}       % 自定义的 MATLAB 高亮，需文件 MATLAB.sty
%\usepackage{pythonhighlight} % Python 高亮，需文件 pythonhighlight.sty

%==================== 表格增强 ====================
\usepackage{booktabs}     % 提供三线表命令 \toprule、\midrule、\bottomrule

%==================== 自定义命令（按需保留） ====================
% \upcite：上标引用（将引用标记放在右上角）
\newcommand{\upcite}[1]{\textsuperscript{\textsuperscript{\cite{#1}}}}

% \tabincell：在表格单元格内嵌套一个表格，用于多行文本
\newcommand{\tabincell}[2]{\begin{tabular}{@{}#1@{}}#2\end{tabular}}

% \HRule：一条占满宽度的横线（可用于封面）
\newcommand{\HRule}{\rule{\linewidth}{0.5mm}}

% 以下 \tikzbox 用于给文字加一个带边框的盒子（需要 tikz 支持，若不需要可注释）
\usepackage{tikz}
\usetikzlibrary{shapes,arrows,positioning}
\newlength\tikzboxwidth
\newlength\tikzboxheight
\newcommand\tikzbox[1]{%
    \settowidth\tikzboxwidth{#1}%
    \settoheight\tikzboxheight{#1}%
    \begin{tikzpicture}
        \path[use as bounding box]
            (-0.5\tikzboxwidth,-0.5\tikzboxheight) rectangle
            (0.5\tikzboxwidth,0.5\tikzboxheight);
        \node[inner sep=\tabcolsep+0.5\arrayrulewidth, line width=0.5mm, draw=black]
            at(0,0){#1};
    \end{tikzpicture}%
}

\input{body/cover}

\tableofcontents

\include{body/HW2}

\end{document}
